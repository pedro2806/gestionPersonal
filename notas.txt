CREATE TABLE IF NOT EXISTS `cursos` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `nombre_curso` VARCHAR(150) NOT NULL,
  `descripcion` TEXT NULL,
  `horas` INT NOT NULL DEFAULT 0,
  `institucion` VARCHAR(150) NOT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb3 COLLATE=utf8mb3_spanish_ci;

CREATE TABLE IF NOT EXISTS `usuario_cursos` (
  `id` INT NOT NULL AUTO_INCREMENT,
  `id_usuario` INT NOT NULL,
  `id_curso` INT NOT NULL,
  `fecha_finalizacion` DATE NOT NULL,
  `calificacion` VARCHAR(10) DEFAULT 'N/A',
  `estatus_curso` VARCHAR(30) NOT NULL, -- 'En Curso', 'Completado', 'Pendiente'
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb3 COLLATE=utf8mb3_spanish_ci;

CREATE TABLE IF NOT EXISTS `expediente_documentos` (

  `id_usuario` INT NOT NULL, -- Relacionado con tu tabla general 'usuarios'
